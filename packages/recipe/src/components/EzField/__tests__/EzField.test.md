```jsx
<EzFormLayout style={{maxWidth: '360px'}}>
  <EzLayout layout="basic">
    <EzField
      label="Send to"
      touched
      error="A valid send to email address is required"
      value="Kevi"
      autoFocus
    />
    <EzField label="Reply to" value="ky@sa-pa.com-test" />
```

```jsx
    base: 'stack',
```
